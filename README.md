# taichi-opengl-nbody

- pygame + modergl rendering backend:

python3 main.py --arch=cpu --fps=60

python3 main.py --arch=gpu --fps=0

- Taichi GGUI backend:

python3 main_taichi_ggui.py --arch=cpu --body=256 --fps=0

python3 main_taichi_ggui.py --arch=gpu --body=1024 --fps=0
